179    def get_coordinate_transformation_matrix(self, shape, scale, rotate):
180        """
181        图像变换后，点的变换前后位置的真实变换矩阵
182        不同于 cv2.getRotationMatrix2D 和 cv2.warpPerspective等
183        为基于原点(0, 0)的坐标系
184        ** 此函数矩阵始终以图像左上为原点的坐标系
185
186        Args:
187            shape: h, w
188            scale:
189            rotate:
190
191        Returns:
192
193        """
194        # 中心旋转的变换矩阵
195        mat_scale_rotate = self.scale_rotate2mat(scale, rotate)
196        mat_center_f = self._matrix_eye_offset(-shape[1] / 2, -shape[0] / 2)
197
198        # 获得变换后的图像尺寸及offset
199        x, y, _, _ = self.get_scale_rotate_shape(shape, scale, rotate)
200        mat_offset = self._matrix_eye_offset(x / 2, y / 2)
201
202        # 最终变换矩阵
203        mat_result = mat_offset * mat_scale_rotate * mat_center_f
204        return mat_result

206    def get_scale_rotate_shape(self, shape, scale, rotate):
207        """
208        得到旋转缩放后的图像尺度大小
209        Args:
210            shape: h, w
211            scale:
212            rotate:
213
214        Returns:
215            x, y
216        """
217        mat = self.scale_rotate2mat(scale, rotate)
218        points = np.array([[0, 0],
219                           [0, shape[0]],
220                           [shape[1], 0],
221                           [shape[1], shape[0]]])
222
223        result = mat[:2, :] @ np.concatenate([points, np.ones((points.shape[0], 1))], axis=1).transpose(1, 0)
224        x = result[0, :].max() - result[0, :].min()
225        y = result[1, :].max() - result[1, :].min()
226
227        x_d, y_d = np.min(np.array(result), axis=1)
228
229        return x, y, x_d, y_d

231    @staticmethod
232    def scale_rotate2mat(
233            scale: Union[int, float, List, Tuple] = 1,
234            rotate: Union[int, float] = 0,
235            offset: Union[List, Tuple] = None,
236            rotate_first_flag: bool = True
237    ) -> np.matrix:
238        """
239        先缩放、旋转 后平移的矩阵变换
240        Args:
241            scale:
242            rotate:
243            offset: [x, y] 默认最后做
244            rotate_first_flag:
245                -- 若scale为两个方向不同值，则会出现旋转的优先级问题，即先做旋转还是先做缩放，
246                两个方法所得到的变换矩阵不一样，具体体现在旋转会改变轴的方向，使得缩放的程度会不一样
247
248        Returns:
249
250        """
251        if isinstance(scale, (int, float)):
252            scale_x = scale_y = scale
253        else:
254            scale_x, scale_y = scale
255
256        mat_scale = np.mat([[scale_x, 0, 0],
257                            [0, scale_y, 0],
258                            [0, 0, 1]])
259
260        mat_rotate = np.mat([[np.cos(np.radians(rotate)), -np.sin(np.radians(rotate)), 0],
261                             [np.sin(np.radians(rotate)), np.cos(np.radians(rotate)), 0],
262                             [0, 0, 1]])
263
264        if offset is not None:
265            mat_offset = np.mat([[1, 0, offset[0]],
266                                 [0, 1, offset[1]],
267                                 [0, 0, 1]])
268
269            if rotate_first_flag:
270                mat = mat_offset * mat_scale * mat_rotate
271            else:
272                mat = mat_offset * mat_rotate * mat_scale
273        else:
274            if rotate_first_flag:
275                mat = mat_scale * mat_rotate
276            else:
277                mat = mat_rotate * mat_scale
278
279        return mat

338    def transform_points(**kwargs):
339        """
340        角点翻转
341        Args:
342            **kwargs:
343                points:
344                shape:
345                flip: 0 | 1, Y axis flip and X axis flip
346
347        Returns:
348
349        """
350
351        points = kwargs.get("points", None)
352        shape = kwargs.get("shape", None)
353
354        if points is None or shape is None:
355            return
356
357        flip = kwargs.get("flip", None)
358        if flip == 0:
359            points[:, 0] = shape[1] - points[:, 0]
360            points = points[[3, 2, 1, 0], :]
361        elif flip == 1:
362            points[:, 1] = shape[0] - points[:, 1]
363            points = points[[1, 0, 3, 2], :]
364        else:
365            raise ValueError
366
367        return points

450        """
451        计算矩阵相乘后的累加和
452        """
453        res = 0

455        for i in prange(h):
456            for j in range(w):
457                res += a[i][j] * b[i][j]

461def transform_points(
462        points: np.ndarray,
463        src_shape: Tuple[int, int],
464        scale: Union[int, float, list, tuple] = 1,
465        rotation: Union[float, int] = 0,
466        offset: tuple = (0, 0),
467        flip: int = -1
468) -> [np.ndarray, np.matrix]:
469    """
470
471    Args:
472        points: n * 2/4大小数组 -- (x, y)
473        src_shape: 原始图像尺寸 -- (h, w)
474        scale:
475        rotation:
476        offset: 该值定义为做完所有变换操作后再操作 -- (x, y)
477        flip: -1 表示不做  0为x坐标翻转 1为y坐标翻转
478
479    Returns:
480        new_points:
481        mat: 仅为scale和rotate的变换矩阵
482
483    """
484    align = Alignment()
485
486    src_shape = list(map(lambda x: x - 1, src_shape))
487
488    mat = align.get_coordinate_transformation_matrix(shape=src_shape, scale=scale, rotate=rotation)
489    if flip == 0:
490        points[:, 0] = src_shape[1] - points[:, 0]
491    elif flip == 1:
492        points[:, 1] = src_shape[0] - points[:, 1]
493    else:
494        pass
495
496    new_points = align.get_points_by_matrix(points, mat)
497
498    new_points = new_points + offset
499
500    p = copy.copy(points)
501    p[:, :2] = new_points
502
503    return p, mat
