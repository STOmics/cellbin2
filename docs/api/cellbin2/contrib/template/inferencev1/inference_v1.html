239    def resove_affine_matrix(H):
240        theta = (math.degrees(math.atan2(H[1, 0], H[0, 0])) + math.degrees(math.atan2(H[1, 1], H[0, 1])) - 90) / 2
241        s_x = math.sqrt(H[0, 0] ** 2 + H[1, 0] ** 2)
242        s_y = (H[0, 0] * H[1, 1] - H[1, 0] * H[0, 1]) / s_x
243        confidence = (H[0, 0] * H[0, 1] + H[1, 0] * H[1, 1]) / s_x
244        clog.info("result: \ntheta: {}, \ns_x: {}, \ns_y: {}, \nconfidence:{}".format(theta, s_x, s_y, 1 - abs(confidence)))
245        return theta, s_x, s_y

247    def _mean_to_scale_and_rotate(self, points_re, points_qc):
248        '''求模板点和QC点 scale和rotate 差异的均值'''
249        scale_x_list = []
250        scale_y_list = []
251        rotate_list = []
252        for point_re, point_qc in zip(points_re, points_qc):
253
254            if point_qc[0] != self.template_center_pt[0]:
255
256                #旋转角
257                rotation_dst = math.degrees(
258                    math.atan((point_qc[1] - self.template_center_pt[1]) / (point_qc[0] - self.template_center_pt[0])))
259                rotation_src = math.degrees(
260                    math.atan((point_re[1] - self.template_center_pt[1]) / (point_re[0] - self.template_center_pt[0])))
261
262                _rotate = self.rotation + (rotation_dst - rotation_src)
263
264                src_x = point_re[0] - self.template_center_pt[0]
265                src_y = point_re[1] - self.template_center_pt[1]
266
267                dst_x = point_qc[0] - self.template_center_pt[0]
268                dst_y = point_qc[1] - self.template_center_pt[1]
269
270                dis = math.sqrt(src_x ** 2 + src_y ** 2)
271
272                _src_x = dis * math.cos(math.radians(np.abs(rotation_dst)))
273                _src_y = dis * math.sin(math.radians(np.abs(rotation_dst)))
274
275                _scale_x = self.scale_x / np.abs(_src_x / dst_x)
276                _scale_y = self.scale_y / np.abs(_src_y / dst_y)
277
278                scale_x_list.append(_scale_x)
279                scale_y_list.append(_scale_y)
280                rotate_list.append(_rotate)
281
282        return np.mean(scale_x_list), np.mean(scale_y_list), np.mean(rotate_list)

284    @staticmethod
285    def _leastsq_to_scale_and_rotate(point_re, point_qc, method="nelder-mead"):
286        '''最小化模板点和QC点距离 并求解出结果'''
287        # point_re = np.array([[61.237, 35.355], [-35.355, 61.237], [-61.237, -35.355], [35.355, -61.237]])
288        # point_qc = np.array([[100, 100], [-100, 100], [-100, -100], [100, -100]])
289        from scipy.optimize import leastsq, minimize
290
291        for k, point in enumerate(point_re):
292            if point[0] == 0 and point[1] == 0:
293                point_re = np.delete(point_re, k, axis=0)
294                point_qc = np.delete(point_qc, k, axis=0)
295                break
296
297        if len(point_re) == 0 or len(point_qc) == 0:
298            return None
299
300        def _error(p, point_re, point_qc):
301            _scale_x, _scale_y, _rotate = p
302
303            src_x = point_re[:, 0]
304            src_y = point_re[:, 1]
305
306            _t = (src_y) / (src_x + 0.000000000000001)
307            _d = [math.atan(i) for i in _t]
308            rotation_src = np.array([math.degrees(i) for i in _d])
309
310            src_x = point_re[:, 0] * (1 + _scale_x)
311            src_y = point_re[:, 1] * (1 + _scale_y)
312
313            dis = [math.sqrt(i) for i in src_x ** 2 + src_y ** 2]
314
315            dst_x = dis * np.array([math.cos(math.radians(np.abs(i))) for i in (rotation_src + _rotate)])
316            dst_y = dis * np.array([math.sin(math.radians(np.abs(i))) for i in (rotation_src + _rotate)])
317
318            dst_x = [-i if point_re[k, 0] < 0 else i for k, i in
319                     enumerate(dst_x)]
320            dst_y = [-i if point_re[k, 1] < 0 else i for k, i in
321                     enumerate(dst_y)]
322
323            error = (point_qc[:, 0] - dst_x) ** 2 + (point_qc[:, 1] - dst_y) ** 2
324            error = [math.sqrt(i) for i in error]
325            # print(np.sum(error))
326            return np.sum(error) * 0 + max(error) * 1
327
328        if method == 'nelder-mead':
329            mini_options = {'maxiter': 100}
330        elif method == 'slsqp':
331            mini_options = {'maxiter': 100}
332        elif method == 'bfgs':
333            mini_options = {'maxiter': 4}
334
335        para = minimize(_error, x0=np.zeros(3, ), args=(point_re, point_qc), method=method,
336                        options=mini_options)
337
338        return para

340    def _caculate_scale_and_rotate(self, points_re, points_qc, mode='minimize', update=True, center=None):
341        '''
342        使用模板点和QC点计算出scale和rotate
343        '''
344        if points_re.shape[1] == 4:
345            points_re = points_re[:, :2]
346
347        if center is None:
348            center = self.template_center_pt
349
350        points_re[:, 0] -= center[0]
351        points_re[:, 1] = center[1] - points_re[:, 1]
352
353        points_qc[:, 0] -= center[0]
354        points_qc[:, 1] = center[1] - points_qc[:, 1]
355
356        if points_qc.shape[0] == 1 and \
357                points_qc[0, 0] == 0 and \
358                points_qc[0, 1] == 0:
359            return
360
361        points_re = np.round(points_re, 2)
362        points_qc = np.round(points_qc, 2)
363
364        if mode == 'minimize':
365            # 最小值距离优化法
366            para = self._leastsq_to_scale_and_rotate(points_re, points_qc, self._method)
367            if para is None:
368                return
369            _scale_x, _scale_y, _rotate = para.x
370            if update:
371                self.rotation -= np.around(_rotate, 5)
372                self.scale_x *= (1 + np.around(_scale_x, 5))
373                self.scale_y *= (1 + np.around(_scale_y, 5))
374                # self.rotation -= _rotate
375                # self.scale_x *= (1 + _scale_x)
376                # self.scale_y *= (1 + _scale_y)
377        elif mode == 'mean':
378            # 均值法
379            scale_x, scale_y, rotate = self._mean_to_scale_and_rotate(points_re, points_qc)
380            if update:
381                self.rotation = rotate
382                self.scale_x = scale_x
383                self.scale_y = scale_y
384        elif mode == 'homo':
385            # 单应性法
386            matrix, mask = cv.estimateAffinePartial2D(points_re, points_qc)
387            # matrix, mask = cv.findHomography(points_re, points_qc, 0)
388            theta, s_x, s_y = self.resove_affine_matrix(matrix)
389            if update:
390                self.rotation += theta
391                self.scale_x = self.scale_x * s_x
392                self.scale_y = self.scale_y * s_y

445    def reference_template(self, mode='single'):
446        '''
447        mode: have three type ['single', 'double', 'multi']
448        *   single: only reference template FOV
449        *   double: reference template FOV & minimize the points distance
450        *   multi: reference template FOV & minimize the points distance & change the template center point
451        '''
452        self._check_parm()
453
454        if len(self.template_center_pt) == 0:
455            self.flag_skip_reference = True
456            clog.info("Template reference not applicable, skip this mode.")
457            return
458
459        self._point_inference(self.template_center_pt, (self.mosaic_height, self.mosaic_width))
460        max_item = round(max(self.mosaic_height, self.mosaic_width) / self._range_thresh)
461
462        if mode != 'single':
463            # self._qc_points_to_gloabal()
464            self._global_qc_points_to_global()
465            points_qc = np.zeros([0, 2])
466            count = 0  # 循环次数
467            while count < max_item:
468
469                temp_scale_x = self.scale_x
470                temp_scale_y = self.scale_y
471                temp_rotate = self.rotation
472
473                points_re, points_qc = self.pair_to_template(self.template_qc, self.template, self._pair_thresh)
474
475                points_re, points_qc = self._delete_outline_points(points_re, points_qc,
476                                                                   self._range_thresh * (count + 1))
477                if len(points_re) == 0:
478                    self.flag_skip_reference = True
479                    clog.info("Template reference not applicable, skip this mode.")
480                    return
481                self._caculate_scale_and_rotate(points_re, points_qc)
482
483                if self.scale_x <= 0 or self.scale_y <= 0 or \
484                        np.abs(temp_scale_x - self.scale_x) > 0.1 or \
485                        np.abs(temp_scale_y - self.scale_y) > 0.1 or \
486                        np.abs(temp_rotate - self.rotation) > 5:
487                    self.scale_x = temp_scale_x
488                    self.scale_y = temp_scale_y
489                    self.rotation = temp_rotate
490
491                self._point_inference(self.template_center_pt, (self.mosaic_height, self.mosaic_width))
492                count += 1
493
494            _, _, _, _, first_recall = self.get_template_eval(area_eval=False)
495            double_info = {'dis': first_recall, 'point': self.template_center_pt,
496                           'scale_x': self.scale_x, 'scale_y': self.scale_y,
497                           'rotate': self.rotation}
498
499            if mode == 'multi':

501                candidate_points = self._find_center_close_point(_points_qc, reverse=True)

512                        self._caculate_scale_and_rotate(points_re, points_qc, update=True)

515                                np.abs(double_info['scale_y'] - self.scale_y) > 0.1 or \

768    def _get_intercept(self, scale, intercept0, region, ind, templ):
769        idx = intercept0
770        intercept = [[idx, ind]]
771        s, e = region
772        item_count = len(templ)
773        # face to large
774        while idx < e:
775            ind = int(ind % item_count)
776            item_len = (templ[ind] * scale) / np.cos(np.radians(self.rotation))
777            idx += item_len
778            intercept.append([idx, (ind + 1) % item_count])
779            ind += 1
780        # face to small
781        idx, ind = intercept[0]
782        while idx > s:
783            ind -= 1
784            ind = int(ind % item_count)
785            item_len = (templ[ind] * scale) / np.cos(np.radians(self.rotation))
786            idx -= item_len
787            intercept.append([idx, ind])
788        return sorted(intercept, key=(lambda x: x[0]))

240    def resove_affine_matrix(H):
241        theta = (math.degrees(math.atan2(H[1, 0], H[0, 0])) + math.degrees(math.atan2(H[1, 1], H[0, 1])) - 90) / 2
242        s_x = math.sqrt(H[0, 0] ** 2 + H[1, 0] ** 2)
243        s_y = (H[0, 0] * H[1, 1] - H[1, 0] * H[0, 1]) / s_x
244        confidence = (H[0, 0] * H[0, 1] + H[1, 0] * H[1, 1]) / s_x
245        clog.info("result: \ntheta: {}, \ns_x: {}, \ns_y: {}, \nconfidence:{}".format(theta, s_x, s_y, 1 - abs(confidence)))
246        return theta, s_x, s_y

248    def _mean_to_scale_and_rotate(self, points_re, points_qc):
249        '''求模板点和QC点 scale和rotate 差异的均值'''
250        scale_x_list = []
251        scale_y_list = []
252        rotate_list = []
253        for point_re, point_qc in zip(points_re, points_qc):
254
255            if point_qc[0] != self.template_center_pt[0]:
256
257                #旋转角
258                rotation_dst = math.degrees(
259                    math.atan((point_qc[1] - self.template_center_pt[1]) / (point_qc[0] - self.template_center_pt[0])))
260                rotation_src = math.degrees(
261                    math.atan((point_re[1] - self.template_center_pt[1]) / (point_re[0] - self.template_center_pt[0])))
262
263                _rotate = self.rotation + (rotation_dst - rotation_src)
264
265                src_x = point_re[0] - self.template_center_pt[0]
266                src_y = point_re[1] - self.template_center_pt[1]
267
268                dst_x = point_qc[0] - self.template_center_pt[0]
269                dst_y = point_qc[1] - self.template_center_pt[1]
270
271                dis = math.sqrt(src_x ** 2 + src_y ** 2)
272
273                _src_x = dis * math.cos(math.radians(np.abs(rotation_dst)))
274                _src_y = dis * math.sin(math.radians(np.abs(rotation_dst)))
275
276                _scale_x = self.scale_x / np.abs(_src_x / dst_x)
277                _scale_y = self.scale_y / np.abs(_src_y / dst_y)
278
279                scale_x_list.append(_scale_x)
280                scale_y_list.append(_scale_y)
281                rotate_list.append(_rotate)
282
283        return np.mean(scale_x_list), np.mean(scale_y_list), np.mean(rotate_list)

285    @staticmethod
286    def _leastsq_to_scale_and_rotate(point_re, point_qc, method="nelder-mead"):
287        '''最小化模板点和QC点距离 并求解出结果'''
288        # point_re = np.array([[61.237, 35.355], [-35.355, 61.237], [-61.237, -35.355], [35.355, -61.237]])
289        # point_qc = np.array([[100, 100], [-100, 100], [-100, -100], [100, -100]])
290        from scipy.optimize import leastsq, minimize
291
292        for k, point in enumerate(point_re):
293            if point[0] == 0 and point[1] == 0:
294                point_re = np.delete(point_re, k, axis=0)
295                point_qc = np.delete(point_qc, k, axis=0)
296                break
297
298        if len(point_re) == 0 or len(point_qc) == 0:
299            return None
300
301        def _error(p, point_re, point_qc):
302            _scale_x, _scale_y, _rotate = p
303
304            src_x = point_re[:, 0]
305            src_y = point_re[:, 1]
306
307            _t = (src_y) / (src_x + 0.000000000000001)
308            _d = [math.atan(i) for i in _t]
309            rotation_src = np.array([math.degrees(i) for i in _d])
310
311            src_x = point_re[:, 0] * (1 + _scale_x)
312            src_y = point_re[:, 1] * (1 + _scale_y)
313
314            dis = [math.sqrt(i) for i in src_x ** 2 + src_y ** 2]
315
316            dst_x = dis * np.array([math.cos(math.radians(np.abs(i))) for i in (rotation_src + _rotate)])
317            dst_y = dis * np.array([math.sin(math.radians(np.abs(i))) for i in (rotation_src + _rotate)])
318
319            dst_x = [-i if point_re[k, 0] < 0 else i for k, i in
320                     enumerate(dst_x)]
321            dst_y = [-i if point_re[k, 1] < 0 else i for k, i in
322                     enumerate(dst_y)]
323
324            error = (point_qc[:, 0] - dst_x) ** 2 + (point_qc[:, 1] - dst_y) ** 2
325            error = [math.sqrt(i) for i in error]
326            # print(np.sum(error))
327            return np.sum(error) * 0 + max(error) * 1
328
329        if method == 'nelder-mead':
330            mini_options = {'maxiter': 100}
331        elif method == 'slsqp':
332            mini_options = {'maxiter': 100}
333        elif method == 'bfgs':
334            mini_options = {'maxiter': 4}
335
336        para = minimize(_error, x0=np.zeros(3, ), args=(point_re, point_qc), method=method,
337                        options=mini_options)
338
339        return para

341    def _caculate_scale_and_rotate(self, points_re, points_qc, mode='minimize', update=True, center=None):
342        '''
343        使用模板点和QC点计算出scale和rotate
344        '''
345        if points_re.shape[1] == 4:
346            points_re = points_re[:, :2]
347
348        if center is None:
349            center = self.template_center_pt
350
351        points_re[:, 0] -= center[0]
352        points_re[:, 1] = center[1] - points_re[:, 1]
353
354        points_qc[:, 0] -= center[0]
355        points_qc[:, 1] = center[1] - points_qc[:, 1]
356
357        if points_qc.shape[0] == 1 and \
358                points_qc[0, 0] == 0 and \
359                points_qc[0, 1] == 0:
360            return
361
362        points_re = np.round(points_re, 2)
363        points_qc = np.round(points_qc, 2)
364
365        if mode == 'minimize':
366            # 最小值距离优化法
367            para = self._leastsq_to_scale_and_rotate(points_re, points_qc, self._method)
368            if para is None:
369                return
370            _scale_x, _scale_y, _rotate = para.x
371            if update:
372                self.rotation -= np.around(_rotate, 5)
373                self.scale_x *= (1 + np.around(_scale_x, 5))
374                self.scale_y *= (1 + np.around(_scale_y, 5))
375                # self.rotation -= _rotate
376                # self.scale_x *= (1 + _scale_x)
377                # self.scale_y *= (1 + _scale_y)
378        elif mode == 'mean':
379            # 均值法
380            scale_x, scale_y, rotate = self._mean_to_scale_and_rotate(points_re, points_qc)
381            if update:
382                self.rotation = rotate
383                self.scale_x = scale_x
384                self.scale_y = scale_y
385        elif mode == 'homo':
386            # 单应性法
387            matrix, mask = cv.estimateAffinePartial2D(points_re, points_qc)
388            # matrix, mask = cv.findHomography(points_re, points_qc, 0)
389            theta, s_x, s_y = self.resove_affine_matrix(matrix)
390            if update:
391                self.rotation += theta
392                self.scale_x = self.scale_x * s_x
393                self.scale_y = self.scale_y * s_y
