 59    def _clean_json(self, json_str):
 60        json_str = re.sub(r'(^|[^:])//\s.*', '', json_str, flags=re.MULTILINE)
 61        json_str = re.sub(r',\s*}', '}', json_str)
 62        json_str = re.sub(r',\s*\]', ']', json_str)
 63        json_str = re.sub(r'\}\s*,\s*}', '}}', json_str)
 64        json_str = re.sub(r'\}\s*,\s*]', '}]', json_str)
 65        json_str = re.sub(r'\]\s*,\s*}', ']}', json_str)
 66        json_str = re.sub(r'(?<=[{,])\s*(\w+)\s*:', r'"\1":', json_str)
 67        json_str = re.sub(r':\s*([a-zA-Z0-9_]+)\s*(?=[,}])', r': "\1"', json_str)
 68        return json_str
