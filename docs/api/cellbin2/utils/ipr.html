410def read(file_path: str) -> Tuple[ImageProcessRecord, Dict[str, Union[IFChannel, ImageChannel]]]:
411    dct = {}
412    with h5py.File(file_path, 'r') as fd:
413        h52dict(fd, dct)
414    ipr_dct = {}
415    image_dct: Dict[str, Union[IFChannel, ImageChannel]] = {}
416
417    for k, v in dct.items():
418        if k in ['IPRVersion', 'ManualState', 'Preview', 'StereoResepSwitch']:
419            ipr_dct[k] = v
420        elif 'Calibration' in v.keys():
421            image_dct[k] = objtyping.from_primitive(v, IFChannel)
422        else:
423            image_dct[k] = objtyping.from_primitive(v, ImageChannel)
424    ipr = objtyping.from_primitive(ipr_dct, ImageProcessRecord)
425
426    return ipr, image_dct

429def read_key_metrics(file_path: str) -> dict:
430    dct = {}
431    with h5py.File(file_path, 'r') as fd:
432        h52dict(fd, dct)
433    image_dct = {}
434
435    dct = {k: v for k, v in dct.items() if k not in ['IPRVersion', 'ManualState', 'Preview', 'StereoResepSwitch']}
436    for channel_name, channel_data in dct.items():
437        image_dct[channel_name] = {k: v for k, v in channel_data.items() if k in ['ImageInfo', 'QCInfo', 'Register']}
438
439    return image_dct
